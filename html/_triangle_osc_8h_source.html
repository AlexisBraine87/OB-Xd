97 inline void processSlave(float x , float delta,bool hardSyncReset,float hardSyncFrac)
98 {
99 bool hspass = true;
100 if(x >= 1.0)
101 {
102 x-=1.0;
103 if(((!hardSyncReset)||(x/delta > hardSyncFrac)))//de morgan processed equation
104 {
105 mixInBlampCenter(buffer1,bP1,x/delta,-4*Samples*delta);
106 }
107 else
108 {
109 x+=1;
110 hspass = false;
111 }
112 }
113 if(x >= 0.5 && x - delta < 0.5 &&hspass)
114 {
115 float frac = (x - 0.5) / delta;
116 if(((!hardSyncReset)||(frac > hardSyncFrac)))//de morgan processed equation
117 {
118 mixInBlampCenter(buffer1,bP1,frac,4*Samples*delta);
119 }
120 }
121 if(x >= 1.0 && hspass)
122 {
123 x-=1.0;
124 if(((!hardSyncReset)||(x/delta > hardSyncFrac)))//de morgan processed equation
125 {
126 mixInBlampCenter(buffer1,bP1,x/delta,-4*Samples*delta);
127 }
128 else
129 {
130 //if transition do not ocurred
131 x+=1;
132 }
133 }
134 if(hardSyncReset)
135 {
136 float fracMaster = (delta * hardSyncFrac);
137 float trans = (x-fracMaster);
138 float mix = trans < 0.5 ? 2*trans-0.5 : 1.5-2*trans;
139 if(trans >0.5)
140 mixInBlampCenter(buffer1,bP1,hardSyncFrac,-4*Samples*delta);
141 mixInImpulseCenter(buffer1,bP1,hardSyncFrac,mix+0.5);
142 }
143 }
144 inline void mixInBlampCenter(float * buf,int& bpos,float offset, float scale)
145 {
146 int lpIn =(int)(B_OVERSAMPLING*(offset));
147 float frac = offset * B_OVERSAMPLING - lpIn;
148 float f1 = 1.0f-frac;
149 for(int i = 0 ; i < n;i++)
150 {
151 float mixvalue = (blampPTR[lpIn]*f1+blampPTR[lpIn+1]*(frac));
152 buf[(bpos+i)&(n-1)] += mixvalue*scale;
153 lpIn += B_OVERSAMPLING;
154 }
155 }
156 inline void mixInImpulseCenter(float * buf,int& bpos,float offset, float scale)
157 {
158 int lpIn =(int)(B_OVERSAMPLING*(offset));
159 float frac = offset * B_OVERSAMPLING - lpIn;
160 float f1 = 1.0f-frac;
161 for(int i = 0 ; i < Samples;i++)
162 {
163 float mixvalue = (blepPTR[lpIn]*f1+blepPTR[lpIn+1]*(frac));
164 buf[(bpos+i)&(n-1)] += mixvalue*scale;
165 lpIn += B_OVERSAMPLING;
166 }
167 for(int i = Samples ; i <n;i++)
168 {
169 float mixvalue = (blepPTR[lpIn]*f1+blepPTR[lpIn+1]*(frac));
170 buf[(bpos+i)&(n-1)] -= mixvalue*scale;
171 lpIn += B_OVERSAMPLING;
172 }
173 }
174 inline float getNextBlep(float* buf,int& bpos)
175 {

177 bpos++;
178
179 // Wrap pos
180 bpos&=(n-1);
181 return buf[bpos];
182 }
