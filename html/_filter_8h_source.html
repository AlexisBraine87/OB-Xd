107 inline float Apply(float sample,float g)
108 {
109
110 float gpw = tanf(g *sampleRateInv * juce::float_Pi);
111 g = gpw;
112 //float v = ((sample- R * s1*2 - g2*s1 - s2)/(1+ R*g1*2 + g1*g2));
113 float v = NR(sample,g);
114
115 float y1 = v*g + s1;
116 s1 = v*g + y1;
117
118 float y2 = y1*g + s2;
119 s2 = y1*g + y2;
120
121 float mc;
122 if(!bandPassSw)
123 mc = (1-mm)*y2 + (mm)*v;
124 else
125 {
126
127 mc =2 * ( mm < 0.5 ?
128 ((0.5 - mm) * y2 + (mm) * y1):
129 ((1-mm) * y1 + (mm-0.5) * v)
130 );
131 }
132
133 return mc;
134 }
135 inline float NR24(float sample,float g,float lpc)
136 {
137 float ml = 1 / (1+g);
138 float S = (lpc*(lpc*(lpc*s1 + s2) + s3) +s4)*ml;
139 float G = lpc*lpc*lpc*lpc;
140 float y = (sample - R24 * S) / (1 + R24*G);
141 return y;
142 }
143 inline float Apply4Pole(float sample,float g)
144 {
145 float g1 = (float)tan(g *sampleRateInv * juce::float_Pi);
146 g = g1;
147
148
149
150 float lpc = g / (1 + g);
151 float y0 = NR24(sample,g,lpc);
152 //first low pass in cascade
153 double v = (y0 - s1) * lpc;
154 double res = v + s1;
155 s1 = res + v;
156 //damping
157 s1 =atan(s1*rcor24)*rcor24Inv;
158
159 float y1= res;
160 float y2 = tptpc(s2,y1,g);
161 float y3 = tptpc(s3,y2,g);
162 float y4 = tptpc(s4,y3,g);
163 float mc;
164 switch(mmch)
165 {
166 case 0:
167 mc = ((1 - mmt) * y4 + (mmt) * y3);
168 break;
169 case 1:
170 mc = ((1 - mmt) * y3 + (mmt) * y2);
171 break;
172 case 2:
173 mc = ((1 - mmt) * y2 + (mmt) * y1);
174 break;
175 case 3:
176 mc = y1;
177 break;
178 default:
179 mc=0;
180 break;
181 }
182 //half volume comp
183 return mc * (1 + R24 * 0.45);
184 }
