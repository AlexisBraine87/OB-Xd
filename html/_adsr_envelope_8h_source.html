108 inline float processSample()
109 {
110 switch (state)
111 {
112 case 1:
113 if (Value - 1 > -0.1)
114 {
115 Value = jmin(Value, 0.99f);
116 state = 2;
117 coef = (float)((log(jmin(sustain + 0.0001, 0.99)) - log(1.0)) / (SampleRate * (decay) / 1000));
118 goto dec;
119 }
120 else
121 {
122 Value = Value - (1-Value)*(coef);
123 }
124 break;
125 case 2:
126 dec:
127 if (Value - sustain < 10e-6)
128 {
129 state = 3;
130 }
131 else
132 {
133 Value =Value + Value * coef;
134 }
135 break;
136 case 3: Value = jmin(sustain, 0.9f);
137 break;
138 case 4:
139 if (Value > 20e-6)
140 Value = Value + Value * coef + dc;
141 else state = 5;
142 break;
143 case 5:
144 Value = 0.0f;
145 break;
146 }
147 return Value;
148 }
