101 inline void mixInImpulseCenter(float * buf,int& bpos,float offset, float scale)
102 {
103 int lpIn =(int)(B_OVERSAMPLING*(offset));
104 float frac = offset * B_OVERSAMPLING - lpIn;
105 float f1 = 1.0f-frac;
106 for(int i = 0 ; i < Samples;i++)
107 {
108 float mixvalue = (blepPTR[lpIn]*f1+blepPTR[lpIn+1]*frac);
109 buf[(bpos+i)&(n-1)] += mixvalue*scale;
110 lpIn += B_OVERSAMPLING;
111 }
112 for(int i = Samples ; i <n;i++)
113 {
114 float mixvalue = (blepPTR[lpIn]*f1+blepPTR[lpIn+1]*frac);
115 buf[(bpos+i)&(n-1)] -= mixvalue*scale;
116 lpIn += B_OVERSAMPLING;
117 }
118 }
119 inline float getNextBlep(float* buf,int& bpos)
120 {
121 buf[bpos]= 0.0f;
122 bpos++;
123
124 // Wrap pos
125 bpos&=(n-1);
126 return buf[bpos];
127 }
