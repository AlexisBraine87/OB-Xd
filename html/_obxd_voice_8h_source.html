178 inline float ProcessSample()
179 {
180 double tunedMidiNote = tuning->tunedMidiNote(midiIndx);
181
182 //portamento on osc input voltage
183 //implements rc circuit
184 float ptNote = tptlpupw(prtst, tunedMidiNote - 81, porta * (1+PortaDetune*PortaDetuneAmt),sampleRateInv);
185 osc.notePlaying = ptNote;
186 //both envelopes and filter cv need a delay equal to osc internal delay
187 float lfoDelayed = lfod.feedReturn(lfoIn);
188 //filter envelope undelayed
189 float envm = fenv.processSample() * (1 - (1-velocityValue)*vflt);
190 if(invertFenv)
191 envm = -envm;
192 //filter exp cutoff calculation
193 float cutoffcalc = jmin(
194 getPitch(
195 (lfof?lfoDelayed*lfoa1:0)+
196 cutoff+
197 FltDetune*FltDetAmt+
198 fenvamt*fenvd.feedReturn(envm)+
199 -45 + (fltKF*(ptNote+40))
200 )
201 //noisy filter cutoff
202 +(ng.nextFloat()-0.5f)*3.5f
203 , (flt.SampleRate*0.5f-120.0f));//for numerical stability purposes
204
205 //limit our max cutoff on self osc to prevent alising
206 if(selfOscPush)
207 cutoffcalc = jmin(cutoffcalc,19000.0f);
208
209
210 //PW modulation
211 osc.pw1 = (lfopw1?(lfoIn * lfoa2):0) + (pwEnvBoth?(pwenvmod * envm) : 0);
212 osc.pw2 = (lfopw2?(lfoIn * lfoa2):0) + pwenvmod * envm + pwOfs;
213
214 //Pitch modulation
215 osc.pto1 = (!pitchWheelOsc2Only? (pitchWheel*pitchWheelAmt):0 ) + ( lfoo1?(lfoIn * lfoa1):0) + (pitchModBoth?(envpitchmod * envm):0) + lfoVibratoIn;
216 osc.pto2 = (pitchWheel *pitchWheelAmt) + (lfoo2?lfoIn*lfoa1:0) + (envpitchmod * envm) + lfoVibratoIn;
217
218
219
220 //variable sort magic - upsample trick
221 float envVal = lenvd.feedReturn(env.processSample() * (1 - (1-velocityValue)*vamp));
222
223 float oscps = osc.ProcessSample() * (1 - levelDetuneAmt*levelDetune);
224
225
226 oscps = oscps - tptlpupw(c1,oscps,12,sampleRateInv);
227
228 float x1 = oscps;
229 x1 = tptpc(d2,x1,brightCoef);
230 if(fourpole)
231 x1 = flt.Apply4Pole(x1,(cutoffcalc));
232 else
233 x1 = flt.Apply(x1,(cutoffcalc));
234 x1 *= (envVal);
235 return x1;
236 }

278 void NoteOn(int mididx,float velocity)
279 {
280 if(!shouldProcessed)
281 {
282 //When your processing is paused we need to clear delay lines and envelopes
283 //Not doing this will cause clicks or glitches
284 lenvd.fillZeroes();
285 fenvd.fillZeroes();
286 ResetEnvelope();
287 }
288 shouldProcessed = true;
289 if(velocity!=-0.5)
290 velocityValue = velocity;
291 midiIndx = mididx;
292 if((!Active)||(legatoMode&1))
293 env.triggerAttack();
294 if((!Active)||(legatoMode&2))
295 fenv.triggerAttack();
296 Active = true;
297 }
298 void NoteOff()
299 {
300 if(!sustainHold)
301 {
302 env.triggerRelease();
303 fenv.triggerRelease();
304 }
305 Active = false;
306 }
