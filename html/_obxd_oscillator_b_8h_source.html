164 inline float ProcessSample()
165 {
166 float noiseGen = wn.nextFloat()-0.5;
167 pitch1 = getPitch(dirt * noiseGen + notePlaying + (quantizeCw?((int)(osc1p)):osc1p)+ pto1 + tune + oct+totalDetune*osc1Factor);
168 bool hsr = false;
169 float hsfrac=0;
170 float fs = jmin(pitch1*(sampleRateInv),0.45f);
171 x1+=fs;
172 hsfrac = 0;
173 float osc1mix=0.0f;
174 float pwcalc =jlimit<float>(0.1f,1.0f,(pulseWidth + pw1)*0.5f + 0.5f);
175
176 if(osc1Pul)
177 o1p.processMaster(x1,fs,pwcalc,pw1w);
178 if(osc1Saw)
179 o1s.processMaster(x1,fs);
180 else if(!osc1Pul)
181 o1t.processMaster(x1,fs);
182
183 if(x1 >= 1.0f)
184 {
185 x1-=1.0f;
186 hsfrac = x1/fs;
187 hsr = true;
188 }
189
190
191
192 pw1w = pwcalc;
193
194 hsr &= hardSync;
195 //Delaying our hard sync gate signal and frac
196 hsr = syncd.feedReturn(hsr) != 0.0f;
197 hsfrac = syncFracd.feedReturn(hsfrac);
198
199 if(osc1Pul)
200 osc1mix += o1p.getValue(x1,pwcalc) + o1p.aliasReduction();
201 if(osc1Saw)
202 osc1mix += o1s.getValue(x1) + o1s.aliasReduction();
203 else if(!osc1Pul)
204 osc1mix = o1t.getValue(x1) + o1t.aliasReduction();
205 //Pitch control needs additional delay buffer to compensate
206 //This will give us less aliasing on xmod
207 //Hard sync gate signal delayed too
208 noiseGen = wn.nextFloat()-0.5;
209 pitch2 = getPitch(cvd.feedReturn(dirt *noiseGen + notePlaying + osc2Det + (quantizeCw?((int)(osc2p)):osc2p) + pto2+ osc1mix *xmod + tune + oct +totalDetune*osc2Factor));
210
211 fs = jmin(pitch2 * (sampleRateInv),0.45f);
212
213 pwcalc = jlimit<float>(0.1f,1.0f,(pulseWidth + pw2)*0.5f + 0.5f);
214
215 float osc2mix=0.0f;
216
217 x2 +=fs;
218
219 if(osc2Pul)
220 o2p.processSlave(x2,fs,hsr,hsfrac,pwcalc,pw2w);
221 if(osc2Saw)
222 o2s.processSlave(x2,fs,hsr,hsfrac);
223 else if(!osc2Pul)
224 o2t.processSlave(x2,fs,hsr,hsfrac);
225
226
227 if(x2 >= 1.0f)
228 x2-=1.0;
229
230
231 pw2w=pwcalc;
232 //On hard sync reset slave phase is affected that way
233 if(hsr)
234 {
235 float fracMaster = (fs * hsfrac);
236 x2 =fracMaster;
237 }
238 //Delaying osc1 signal
239 //And getting delayed back
240 osc1mix = xmodd.feedReturn(osc1mix);
241
242 if(osc2Pul)
243 osc2mix += o2p.getValue(x2,pwcalc) + o2p.aliasReduction();
244 if(osc2Saw)
245 osc2mix += o2s.getValue(x2) + o2s.aliasReduction();
246 else if(!osc2Pul)
247 osc2mix = o2t.getValue(x2) + o2t.aliasReduction();
248
249 //mixing
250 float res =o1mx*osc1mix + o2mx *osc2mix + (noiseGen)*(nmx*1.3 + 0.0006);
251 return res*3;
252 }
