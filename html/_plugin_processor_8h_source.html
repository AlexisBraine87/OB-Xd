91// Compares a magic value in either endianness.
92static inline bool compareMagic (int32 magic, const char* name) noexcept
93{
94 return magic == (int32) ByteOrder::littleEndianInt (name)
95 || magic == (int32) ByteOrder::bigEndianInt (name);
96}
97
98static inline int32 fxbName (const char* name) noexcept { return (int32) ByteOrder::littleEndianInt (name); }
99static inline int32 fxbSwap (const int32 x) noexcept { return (int32) ByteOrder::swapIfLittleEndian ((uint32) x); }
100
101static inline float fxbSwapFloat (const float x) noexcept
102{
103#ifdef JUCE_LITTLE_ENDIAN
104 union { uint32 asInt; float asFloat; } n;
105 n.asFloat = x;
106 n.asInt = ByteOrder::swap (n.asInt);
107 return n.asFloat;
108#else
109 return x;
110#endif
111}
