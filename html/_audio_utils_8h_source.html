55inline static float tptlpupw(float & state , float inp , float cutoff , float srInv)
56{
57 cutoff = (cutoff * srInv)*juce::float_Pi;
58 double v = (inp - state) * cutoff / (1 + cutoff);
59 double res = v + state;
60 state = res + v;
61 return res;
62}
