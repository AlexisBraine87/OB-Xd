369 void setXml( XmlElement &xml){
370 for (int i = 0; i < 255; ++i)
371 {
372 if (controllers[i]!= -1){
373 xml.setAttribute("MIDI_" +String(i), getTag(controllers[i]));
374 }
375 }
376 }
377 String getTag(int paraId){
378 for (std::map<String, int>::iterator it = this->mapping.begin(); it != this->mapping.end(); it ++){
379 if (paraId == it->second){
380 return it->first;
381 }
382 }
383 return "undefine";
384 }
385
386 int getParaId(String tagName){
387 return mapping[tagName];
388 }
389
390 void getXml(XmlElement &xml){
391 for (int i = 0; i < 255; ++i)
392 {
393 String tmp = xml.getStringAttribute("MIDI_" + String(i), "undefine");
394 if (tmp != "undefine"){
395 controllers[i] = getParaId(tmp);
396 }
397 }
398
399 // Backward keys
400 if (controllers[100] > 0) {
401 controllers[77] = controllers_default[77];
402 controllers[100] = 0;
403 }
404 if ( controllers[101] > 0) {
405 controllers[78] = controllers_default[78];
406 controllers[101] = 0;
407 }
408 }

410 bool loadFile(File& xml){
411 reset();
412 set_default();
413 if (xml.existsAsFile()){
414 XmlDocument xmlDoc (xml);
415 this->getXml(*xmlDoc.getDocumentElement());
416 return true;
417 }
418
419 return false;
420 }

423 for (int i =0; i < 255; i++) {
